        function calculatePrognosis() {

            // Get input values

            const ageScore = parseInt(document.getElementById('ageGroup').value);

            const genderScore = parseInt(document.getElementById('gender').value);

            const metastasisScore = parseInt(document.getElementById('metastasis').value);

            const stageScore = parseInt(document.getElementById('stage').value);

           

            // Calculate total score

            const totalScore = ageScore + genderScore + metastasisScore + stageScore;

           

            // Risk stratification

            let riskLevel, riskColor, survivalData, recommendation;

            if (totalScore <= 50) {

                riskLevel = "Low Risk";

                riskColor = "#27ae60";

                survivalData = [95, 90, 85, 80, 75];

                recommendation = `

                    <h3>Clinical Recommendations - Low Risk</h3>

                    <ul>

                        <li><strong>Standard treatment protocol recommended</strong></li>

                        <li><strong>Routine follow-up</strong> (6-12 months)</li>

                        <li><strong>Favorable prognosis expected</strong></li>

                        <li><strong>Approx. 65%</strong> 5-year survival rate</li>

                        <li>Consider lower intensity surveillance</li>

                    </ul>

                `;

            } else if (totalScore <= 120) {

                riskLevel = "Medium Risk";

                riskColor = "#f39c12";

                survivalData = [85, 75, 65, 55, 45];

                recommendation = `

                    <h3>Clinical Recommendations - Medium Risk</h3>

                    <ul>

                        <li><strong>Consider adjuvant therapy intensification</strong></li>

                        <li><strong>Close monitoring</strong> (3-6 month intervals)</li>

                        <li><strong>Personalized treatment adjustment</strong></li>

                        <li><strong>Approx. 45%</strong> 5-year survival rate</li>

                        <li>Regular imaging and laboratory follow-up</li>

                    </ul>

                `;

            } else {

                riskLevel = "High Risk";

                riskColor = "#e74c3c";

                survivalData = [70, 55, 40, 30, 20];

                recommendation = `

                    <h3>Clinical Recommendations - High Risk</h3>

                    <ul>

                        <li><strong>Intensified treatment regimen recommended</strong></li>

                        <li><strong>Frequent monitoring</strong> (1-3 month intervals)</li>

                        <li><strong>Consider clinical trial participation</strong></li>

                        <li><strong>Aggressive supportive care</strong></li>

                        <li><strong>Approx. 20%</strong> 5-year survival rate</li>

                        <li>Multidisciplinary team management essential</li>

                    </ul>

                `;

            }

           

            // Extract survival rates

            const oneYearSurvival = survivalData[0];

            const threeYearSurvival = survivalData[2];

            const fiveYearSurvival = survivalData[4];

           

            // Update display

            document.getElementById('scoreDisplay').textContent = totalScore;

            document.getElementById('scoreDisplay').style.color = riskColor;

            document.getElementById('oneYearDisplay').textContent = oneYearSurvival + '%';

            document.getElementById('threeYearDisplay').textContent = threeYearSurvival + '%';

            document.getElementById('fiveYearDisplay').textContent = fiveYearSurvival + '%';

            document.getElementById('riskLevelDisplay').textContent = riskLevel;

            document.getElementById('riskLevelDisplay').style.color = riskColor;

           

            // Update risk marker position (0-270 points mapped to 0-100%)

            const markerPosition = (totalScore / 270) * 100;

            document.getElementById('riskMarker').style.left = markerPosition + '%';

           

            // Update factor contributions chart

            factorsChart.data.datasets[0].data = [ageScore, genderScore, metastasisScore, stageScore];

            factorsChart.update();

           

            // Update survival curve

            survivalChart.data.datasets[0].data = survivalData;

            survivalChart.update();

           

            // Update clinical recommendations

            document.getElementById('clinical-recommendations').innerHTML = recommendation;

            document.getElementById('clinical-recommendations').className = 'clinical-recommendations ' +

                (riskLevel === "Low Risk" ? "low-risk" :

                 riskLevel === "Medium Risk" ? "medium-risk" : "high-risk");

            document.getElementById('clinical-recommendations').style.display = 'block';

           

            // Show results panel

            document.getElementById('results').style.display = 'block';

            document.getElementById('placeholder').style.display = 'none';

        }

       

        // Initialize charts when page loads

        document.addEventListener('DOMContentLoaded', function() {

            initCharts();

        });
